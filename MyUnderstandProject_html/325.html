756         logical function CheckData(self)
757         
758             implicit none;        
759             !********************************************
760             !DECLARATIVE ZONE
761             !********************************************
762             class(DDDMS_DataLoaderFromFile) :: self;
763             character(len = 200)            :: currentLine;
764             integer(INT32)                  :: indexPath;
765             integer(INT32)                  :: countLinesFile;
766             integer(INT32)                  :: status;
767             integer(INT64)                  :: file_size;
768             integer(INT64)                  :: NumOfZones;
769             integer(INT32)                  :: i;
770             integer(INT32)                  :: fid;
771             logical                         :: file_exists; 
772             
773             !********************************************
774             !BODY OF THE PROGRAM
775             !******************************************** 
776             
777             !********************************************************************************************************
778             !CHECK IF THE ENTERED FILE NAME self%inputFilePathName EXISTS AND COUNTS HOW MANY LINES THERE ARE INSIDE IT
779             !********************************************************************************************************
780             inquire(FILE = self%inputFilePathName , EXIST = file_exists); !CHECK IF THE INPUT FILE "SELF%inputFilePathName" EXISTS.
781             inquire(FILE = self%inputFilePathName , SIZE  = file_size);   !CHECK IF THE INPUT FILE "SELF%inputFilePathName" IS EMPTY.

783             indexPath = scan( self%inputFilePathName, "\", BACK = .TRUE.);

785             if( file_exists .AND. (file_size /= 0) ) then
786                 CheckData = .TRUE.;
787                 open(NEWUNIT = fid, ACTION = 'read', FILE = self%inputFilePathName, STATUS = 'old');
788

790                     read(fid, '(A)', IOSTAT = status) currentLine;
791                     if (status == IOSTAT_END) then
792                         exit; ! END OF FILE REACHED.ABM
793                     end if
794                     
795                     if( ( index( trim(currentLine), '.txt') /= 0 ) .OR. &
796                         ( index( trim(currentLine), '.c')   /= 0 ) .OR. & 
797                         ( index( trim(currentLine), '.d')   /= 0 ) )then !THE NAME OF FILE WAS FOUND

805                             CheckData = .FALSE.;
806                             
807                         end if

812             else
813                 write(OUTPUT_UNIT, *) "File name entered does not exist";
814                 write(OUTPUT_UNIT, *) "or its size is equal to zero!";

816                 stop;
817             end if
